# Load libaries -----------------------------------------------------------

library(dplyr)

library(Seurat)

library(ggplot2)

library(UpSetR)



# load loom file ----

lfile <- connect(filename = "l5_all.loom", mode = "r+")

lfile


# loom to seurat ----

adol <- as.Seurat(x = lfile)


# Normalizing the data ----------------------------------------------------

adol <- NormalizeData(object = adol, normalization.method = "LogNormalize", scale.factor = 10000)


# Find variable genes -----------------------------------------------------

adol <- FindVariableFeatures(object = adol, selection.method = "vst")


# Scaling the data --------------------------------------------------------

adol <- ScaleData(object = adol)


# Dimensional reduction ---------------------------------------------------

adol <- RunPCA(object = adol, features = VariableFeatures(object = adol), npcs = 100)


# Determine dimensionality of the data ------------------------------------

ElbowPlot(object = adol, ndims = 100)


# Clustering --------------------------------------------------------------

adol <- FindNeighbors(adol, dims = 1:50)

adol <- FindClusters(object = adol, resolution = 5)

adol <- RunUMAP(object = adol, dims = 1:50, seed.use = 42) 

DimPlot(object = adol, reduction = "umap", label = F) + NoLegend()


# Subset neurons --------------------------------------------------------------

neurons <- subset(adol, subset = Slc17a7 > 1 | Slc17a6 > 1 | Slc17a8 > 1 | Gad1 > 1 | Gad2 > 1 | Slc18a3 > 1 | Chat > 1 | Slc18a2 > 1 | Th > 1 | Slc6a5 > 1 )

FeaturePlot(neurons, features = c("Gad1","Slc17a7", "Slc17a6", "Slc17a8", "Slc18a3", "Chat", "Slc18a2", "Slc6a5"), label = F, ncol = 4) #Neurons

neurons <- NormalizeData(object = neurons, verbose = FALSE)

neurons <- FindVariableFeatures(object = neurons, selection.method = "vst", nfeatures = 2000)

neurons <- RunPCA(object = neurons, features = VariableFeatures(object = neurons), npcs = 100)

ElbowPlot(object = neurons, ndims = 100)

neurons <- FindNeighbors(neurons, dims = 1:30)

neurons <- FindClusters(object = neurons, resolution = 2)

neurons <- RunUMAP(object = neurons, dims = 1:30)

DimPlot(object = neurons, reduction = "umap", label = F)
